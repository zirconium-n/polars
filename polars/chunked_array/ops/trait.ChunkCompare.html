Expand description

Compare Series and ChunkedArray’s and get a boolean mask that can be used to filter rows.

Example

use polars_core::prelude::*;
fn filter_all_ones(df: &DataFrame) -> Result<DataFrame> {
    let mask = df
    .column("column_a")?
    .eq(1);

    df.filter(&mask)
}
